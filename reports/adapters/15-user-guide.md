## Collection Definition

### Real Syntax (from codebase)

```typescript
import { collection, field, f } from '@deessejs/collections'

const posts = collection({
  slug: 'blog-posts',                    // REQUIRED: unique identifier
  name: 'Blog Posts',                    // optional, auto-generated from slug
  admin: {
    description: 'Blog posts for the site',
    icon: 'article',
  },
  fields: {
    // Field syntax: field({ fieldType: f.type(), options })
    title: field({ fieldType: f.text({ maxLength: 200 }), required: true }),
    slug: field({ fieldType: f.text(), unique: true }),
    content: field({ fieldType: f.richtext() }),
    published: field({ fieldType: f.boolean(), defaultValue: false }),
    status: field({ fieldType: f.select(['draft', 'review', 'published']), defaultValue: 'draft' }),
    authorId: field({ fieldType: f.uuid() }),
    metadata: field({ fieldType: f.json() }),
    views: field({ fieldType: f.number({ min: 0 }), defaultValue: 0 }),
  },
  hooks: {
    beforeCreate: async (ctx) => { /* ... */ },
    afterCreate: async (ctx) => { /* ... */ },
  },
  autoGeneratedFields: { id: true, createdAt: true, updatedAt: true }, // defaults
})
```

### Key Points

1. **Slug is in config object** - `collection({ slug: 'posts', ... })` NOT `collection('posts', {...})`
2. **Fields use `field()` wrapper** - `field({ fieldType: f.text(), required: true })` NOT `f.text().required()`
3. **Options are in config object** - `defaultValue`, `unique`, `indexed`, `required` all inside `field({...})`

## Field Types

### Available Field Types

```typescript
import { field, f } from '@deessejs/collections'

// Text
field({ fieldType: f.text() })
field({ fieldType: f.text({ maxLength: 500 }) })
field({ fieldType: f.richtext() })  // Rich text (stored as text)

// Email and URL
field({ fieldType: f.email() })      // z.string().email()
field({ fieldType: f.url() })        // z.string().url()

// Numbers
field({ fieldType: f.number() })
field({ fieldType: f.number({ min: 0, max: 100 }) })
field({ fieldType: f.decimal(10, 2) })  // precision, scale

// Boolean
field({ fieldType: f.boolean() })

// Dates
field({ fieldType: f.date() })         // z.date()
field({ fieldType: f.timestamp() })    // without timezone
field({ fieldType: f.timestampTz() })  // with timezone

// JSON
field({ fieldType: f.json() })
field({ fieldType: f.jsonb() })

// UUID
field({ fieldType: f.uuid() })         // z.string().uuid()

// Select/Enum
field({ fieldType: f.select(['draft', 'published', 'archived']) })

// Relation
field({ fieldType: f.relation() })    // UUID referencing another collection

// Array
field({ fieldType: f.array(f.text()) })  // array of text

// File
field({ fieldType: f.file() })        // stored as string path
```

### Field Options

```typescript
field({
  fieldType: f.text(),
  required: true,           // cannot be null (default: false)
  defaultValue: 'hello',    // static default
  defaultFactory: () => Math.random(),  // dynamic default
  unique: true,             // unique constraint (default: false)
  indexed: true,            // add database index (default: false)
})
```

### Auto-Generated Fields

By default, collections automatically get:
- `id` - UUID (auto-generated)
- `createdAt` - timestamp
- `updatedAt` - timestamp

To disable:

```typescript
collection({
  slug: 'posts',
  fields: { /* ... */ },
  autoGeneratedFields: {
    id: false,        // disable auto id
    createdAt: false,  // disable createdAt
    updatedAt: false,  // disable updatedAt
  }
})
```

## Querying

### Where Conditions

```typescript
import { where, eq, ne, gt, lt, gte, lte, and, or, not } from '@deessejs/collections'
import { between, inList, notInList } from '@deessejs/collections'
import { isNull, isNotNull } from '@deessejs/collections'
import { like, contains, startsWith, endsWith, regex } from '@deessejs/collections'
import { has, hasAny, overlaps } from '@deessejs/collections'
import { search } from '@deessejs/collections'

// Equality
where((p) => [eq(p.status, 'published')])

// Not equals
where((p) => [ne(p.status, 'archived')])

// Comparison
where((p) => [gt(p.views, 100)])
where((p) => [lt(p.createdAt, newDate)])
where((p) => [gte(p.priority, 5)])
where((p) => [lte(p.score, 100)])

// Between
where((p) => [between(p.views, 100, 500)])

// In list
where((p) => [inList(p.status, ['draft', 'review'])])

// Null checks
where((p) => [isNull(p.deletedAt)])
where((p) => [isNotNull(p.publishedAt)])

// Text search (LIKE)
where((p) => [like(p.title, '%typescript%')])
where((p) => [contains(p.title, 'guide')])        // LIKE '%value%'
where((p) => [startsWith(p.slug, 'how-to-')])      // LIKE 'value%'
where((p) => [endsWith(p.email, '@company.com')])  // LIKE '%value'

// Regex (database-specific)
where((p) => [regex(p.title, '^[a-z]')])

// Array operators
where((p) => [has(p.tags, 'featured')])
where((p) => [hasAny(p.tags, ['featured', 'promoted'])])
where((p) => [overlaps(p.tags, ['a', 'b'])])

// Full-text search
where((p) => [search(['title', 'content'], 'typescript')])

// Multiple conditions (AND)
where((p) => [
  eq(p.published, true),
  gt(p.views, 10)
])

// OR - use or() function
where((p) => [
  or(
    eq(p.status, 'published'),
    eq(p.featured, true)
  )
])

// NOT
where((p) => [
  not(eq(p.status, 'archived'))
])
```

### Available Operators

| Operator | Description | Example |
|----------|-------------|---------|
| `eq` | Equals | `eq(p.x, value)` |
| `ne` | Not equals | `ne(p.x, value)` |
| `gt` | Greater than | `gt(p.x, 5)` |
| `gte` | Greater or equal | `gte(p.x, 5)` |
| `lt` | Less than | `lt(p.x, 10)` |
| `lte` | Less or equal | `lte(p.x, 10)` |
| `between` | Between range | `between(p.x, min, max)` |
| `inList` | In array | `inList(p.x, ['a', 'b'])` |
| `notInList` | Not in array | `notInList(p.x, ['a', 'b'])` |
| `isNull` | Is null | `isNull(p.x)` |
| `isNotNull` | Is not null | `isNotNull(p.x)` |
| `like` | SQL LIKE | `like(p.x, '%pattern%')` |
| `contains` | Contains substring | `contains(p.x, 'sub')` |
| `startsWith` | Starts with | `startsWith(p.x, 'prefix')` |
| `endsWith` | Ends with | `endsWith(p.x, 'suffix')` |
| `regex` | Regex match | `regex(p.x, '^pattern')` |
| `has` | Array contains | `has(p.tags, 'featured')` |
| `hasAny` | Array contains any | `hasAny(p.tags, ['a', 'b'])` |
| `overlaps` | Arrays overlap | `overlaps(p.tags, ['a', 'b'])` |
| `search` | Full-text search | `search(['title'], 'query')` |

### Logical Combinators

```typescript
// AND is implicit when array has multiple conditions
where((p) => [eq(p.a, 1), eq(p.b, 2)])  // a = 1 AND b = 2

// OR requires explicit or()
where((p) => [
  or(eq(p.status, 'published'), eq(p.featured, true))
])

// NOT
where((p) => [not(eq(p.status, 'archived'))])
```

### Order By

```typescript
import { orderBy, asc, desc } from '@deessejs/collections'

// Single field
```

## Hooks

### Available Hooks

```typescript
const posts = collection({
  slug: 'posts',
  fields: { /* ... */ },
  hooks: {
    // Operation lifecycle
    beforeOperation: async (ctx) => ctx,
    afterOperation: async (ctx) => ctx,

    // Create
    beforeCreate: async (ctx) => {
      ctx.data.title = ctx.data.title.trim()
      return ctx
    },
    afterCreate: async (ctx) => {
      console.log('Created:', ctx.data)
      return ctx
    },

    // Update
    beforeUpdate: async (ctx) => {
      ctx.data.updatedAt = new Date().toISOString()
      return ctx
    },
    afterUpdate: async (ctx) => ctx,

    // Delete
    beforeDelete: async (ctx) => {
      console.log('Deleting:', ctx.previousData)
      return ctx
    },
    afterDelete: async (ctx) => ctx,

    // Read
    beforeRead: async (ctx) => {
      ctx.query.limit = 10  // modify query
      return ctx
    },
    afterRead: async (ctx) => ctx,
  }
})
```

### Hook Context Types

```typescript
// CreateHookContext
{
  collection: 'posts'
  operation: 'create'
  data: { title: string; content: string; ... }  // mutable
}

// UpdateHookContext
{
  collection: 'posts'
  operation: 'update'
  data: Partial<Fields>      // mutable
  where: Record<string, unknown>
  previousData: Fields        // immutable
}

// DeleteHookContext
{
  collection: 'posts'
  operation: 'delete'
  where: Record<string, unknown>
  previousData: Fields        // immutable
}

// ReadHookContext
{
  collection: 'posts'
  operation: 'read'
  query: { where?, orderBy?, limit?, offset?, select? }  // mutable
}
```

### Early Exit

```typescript
beforeCreate: async (ctx) => {
  if (ctx.data.draft) {
    return { _stop: true, data: ctx.data }
  }
  return ctx
}
```

## Error Handling *(Planned)*

```typescript
import { AdapterError, UniqueConstraintError, NotFoundError } from '@deessejs/collections/adapter'

try {
  await db.posts.create({ data: { title: 'Duplicate' } })
} catch (error) {
  if (error instanceof UniqueConstraintError) {
    console.log(`Field ${error.field} already exists`)
  }
  if (error instanceof NotFoundError) {
    console.log('Record not found')
  }
  if (error instanceof AdapterError) {
    console.log(error.code, error.details)
  }
}
```
