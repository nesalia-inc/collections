# Field to RawColumn Mapper

> **NOTE**: The switch operates on `colType.name` (the column type name from the column-types layer), NOT the field type name. Many `f.*` builders produce different column types (e.g., `f.email()` produces `varchar` column type).

```typescript
// src/adapter/core/mapper.ts

import type { Field } from '../../fields'
import type { RawColumn } from './types'
import { toSnakeCase } from '../../utils/toSnakeCase'

export const fieldToRawColumn = (
  fieldName: string,
  field: Field<unknown>
): RawColumn => {
  const colType = field.fieldType.columnType
  const base = {
    name: toSnakeCase(fieldName),
    notNull: field.required,
  }

  switch (colType.name) {
    case 'serial':
      return { type: 'serial', ...base, primaryKey: true }
    case 'integer':
      return { type: 'integer', ...base, autoIncrement: false }
    case 'numeric':
      return { type: 'numeric', ...base, precision: colType.precision, scale: colType.scale }
    case 'decimal':
      return { type: 'decimal', ...base, precision: colType.precision, scale: colType.scale }
    case 'real':
      return { type: 'real', ...base }
    case 'text':
      return { type: 'text', ...base }
    case 'varchar':
      // f.email(), f.url(), f.file() produce varchar column types
      return { type: 'varchar', ...base, length: colType.length }
    case 'char':
      return { type: 'char', ...base, length: colType.length }
    case 'boolean':
      return { type: 'boolean', ...base }
    case 'date':
      return { type: 'date', ...base }
    case 'timestamp':
      return { type: 'timestamp', ...base }
    case 'timestamptz':
      return { type: 'timestamptz', ...base }
    case 'json':
      return { type: 'json', ...base }
    case 'jsonb':
      return { type: 'jsonb', ...base }
    case 'uuid':
      return { type: 'uuid', ...base, defaultRandom: true }
    case 'enum':
      // f.select() produces enum column type
      return { type: 'enum', ...base, enumName: `${toSnakeCase(fieldName)}_enum`, options: colType.values }
    // Field type aliases (colType.name values from f.* builders that differ from column type names)
    case 'select':
      return { type: 'enum', ...base, enumName: `${toSnakeCase(fieldName)}_enum`, options: colType.values || [] }
    case 'relation':
      return { type: 'uuid', ...base, defaultRandom: true }
    case 'array':
      // f.array() stores as JSON
      return { type: 'json', ...base }
    case 'file':
      return { type: 'text', ...base }
    case 'richtext':
      return { type: 'text', ...base }
    default:
      throw new Error(`Unsupported column type: ${colType.name}`)
  }
}
```

## Schema Builder

```typescript
// src/adapter/core/schemaBuilder.ts

import type { Collection } from '../../collections'
import type { RawTable, RawColumn } from './types'
import { fieldToRawColumn } from './mapper'
import { toSnakeCase } from '../../utils/toSnakeCase'

export const buildRawSchema = (collections: Collection[]): Map<string, RawTable> => {
  const tables = new Map<string, RawTable>()

  for (const coll of collections) {
    const columns: Record<string, RawColumn> = {}

    // Add ID field if autoGeneratedFields.id is true (default)
    if (coll.autoGeneratedFields?.id !== false) {
      columns.id = {
        type: 'uuid',
        name: 'id',
        primaryKey: true,
        defaultRandom: true,
        notNull: true
      }
    }

    // Process each field in ownFields (user-defined, not auto-generated)
    for (const [fieldName, field] of Object.entries(coll.ownFields)) {
      const colName = toSnakeCase(fieldName)
      columns[colName] = fieldToRawColumn(fieldName, field)
    }

    // Add createdAt if autoGeneratedFields.createdAt is true (default)
    if (coll.autoGeneratedFields?.createdAt !== false) {
      columns.created_at = {
        type: 'timestamp',
        name: 'created_at',
        notNull: true,
        defaultNow: true
      }
    }

    // Add updatedAt if autoGeneratedFields.updatedAt is true (default)
    if (coll.autoGeneratedFields?.updatedAt !== false) {
      columns.updated_at = {
        type: 'timestamp',
        name: 'updated_at',
        notNull: true,
        defaultNow: true
      }
    }

    tables.set(coll.slug, {
      name: toSnakeCase(coll.slug),
      columns
    })
  }

  return tables
}
```

## Note on Field Names

In the real codebase:
- Collection fields are defined with camelCase (e.g., `title`, `createdAt`)
- Database columns use snake_case (e.g., `title`, `created_at`)
- The `toSnakeCase()` utility converts between them
- Auto-generated fields (`id`, `created_at`, `updated_at`) are controlled by `autoGeneratedFields` config
