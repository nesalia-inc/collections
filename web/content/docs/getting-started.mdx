## Define Your First Collection

Collections are defined as separate files that export a collection definition:

```ts
// collections/users.ts
import { collection } from '@deessejs/collections'
import { field, text, email } from '@deessejs/collections/fields'

export const users = collection({
  slug: 'users',

  fields: {
    name: field({
      type: text({ min: 2, max: 100 })
    }),

    email: field({
      type: email(),
      unique: true
    })
  }
})
```

## Create Another Collection

```ts
// collections/posts.ts
import { collection } from '@deessejs/collections'
import { field, text, enumField } from '@deessejs/collections/fields'

export const posts = collection({
  slug: 'posts',

  fields: {
    title: field({
      type: text({ min: 3, max: 255 })
    }),

    content: field({
      type: text()
    }),

    status: field({
      type: enumField(['draft', 'published', 'archived'])
    })
  }
})
```

## Configure Your Application

```ts
// config/database.ts
import { defineConfig } from '@deessejs/collections'
import { users } from '../collections/users'
import { posts } from '../collections/posts'

export const { collections, db } = defineConfig({
  database: {
    url: process.env.DATABASE_URL!
  },

  collections: [users, posts]
})
```

## Start Querying

```ts
// Anywhere in your app
import { collections } from './config/database'

// Find all users
const allUsers = await collections.users.findMany()

// Find a specific user
const user = await collections.users.findUnique({
  where: { id: 1 }
})

// Create a new post
const post = await collections.posts.create({
  data: {
    title: 'Hello World',
    content: 'My first post',
    status: 'draft'
  }
})

// Update a post
const updated = await collections.posts.update({
  where: { id: 1 },
  data: {
    title: 'Updated Title'
  }
})
```

## Type Safety

Everything is fully typed based on your collection definitions:

```ts
const user = await collections.users.findUnique({
  where: { id: 1 }
})

// TypeScript knows:
// - user.name is a string
// - user.email is a string
// - user is typed as User | null

if (user) {
  console.log(user.name) // âœ… Fully typed
}
```
